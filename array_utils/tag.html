  SeqSum = append(SeqSum, array[0])

    } else {

    }
  }
  max := SeqSum[0]
  for j := 1; j < len(SeqSum); j++ {
    if SeqSum[j] > max {

}

func optimizationFindMaxSum(array []int) int {
  SeqSum := make([]int, len(array))
  for i := 0; i < len(array); i++ {
    SeqSum[i] = array[i]
  }

  for j := 1; j < len(SeqSum); j++ {
    tmp := SeqSum[j-1]
    if tmp < 0 {
      tmp = 0
    }
    SeqSum[j] += tmp
  }
  max := SeqSum[0]
  for k := 1; k < len(SeqSum); k++ {
    if max < SeqSum[k] {
      max = SeqSum[k]
    }
  }
  return max
}
